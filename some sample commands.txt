#### tailwind (tomado de https://www.auroria.io/blog/using-tailwindcss-framework-with-angular)
 $ npm i tailwindcss postcss-loader postcss-scss postcss-import autoprefixer @angular-builders/custom-webpack -D\n
 $ npx tailwind init
 $ touch webpack.config.js
 $ npm i @fullhuman/postcss-purgecss -D
 $ ng serve
### estructura
 $ ng g m layout --routing=true
 $ ng g m control
 $ ng g m control/error -m=control --routing=true
 $ ng g m control/not-found -m=control --routing=true
 $ ng g m control/access-denied -m=control --routing=true
 $ ng g m layout/default -m=layout --routing=true
 $ ng g m layout/default-layout -m=layout --routing=true
 $ ng g c layout --inlineStyle=true --inlineTemplate=true -m=layout --skipSelector=true --skipTests=true
 $ ng g c layout/default-layout --inlineStyle=true -m=layout/default-layout --skipSelector=true --skipTests=true
 $ ng serve
